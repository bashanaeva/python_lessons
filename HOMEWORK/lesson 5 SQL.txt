ВТОРОЕ ЗАДАНИЕ кол-во дней рождения по дням недели
SELECT 
COUNT(*)
DAYNAME (CONCAT(YEAR(NOW),RIGHT(birthday,6)))
FROM `profile`
ORDER BY birthday DESC
